### toggle_switch.dart
```dart
// width 32, height 18
// Active: colorAccent bg
// Inactive: colorG200 bg
// Thumb: 12×12 white circle, positioned top:3 right:3 (active) or top:3 left:3 (inactive)
// Animate thumb position with AnimatedPositioned
GestureDetector(
  onTap: () => setState(() => isOn = !isOn),
  child: AnimatedContainer(
    duration: Duration(milliseconds: 200),
    width: kToggleWidth,
    height: kToggleHeight,
    decoration: BoxDecoration(
      color: isOn ? colorAccent : colorG200,
      borderRadius: BorderRadius.circular(9),
    ),
    child: Stack(
      children: [
        AnimatedPositioned(
          duration: Duration(milliseconds: 200),
          top: 3,
          right: isOn ? 3 : null,
          left: isOn ? null : 3,
          child: Container(
            width: kToggleThumb,
            height: kToggleThumb,
            decoration: BoxDecoration(color: Colors.white, shape: BoxShape.circle),
          ),
        ),
      ],
    ),
  ),
)
```
